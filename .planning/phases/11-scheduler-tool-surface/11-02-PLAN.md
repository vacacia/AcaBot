    ```python
    async def list_scheduler_tasks(self, *, owner: str) -> dict[str, Any]:
        """列出指定 owner 的所有定时任务.

        Args:
            owner: 任务所有者 session_id.

        Returns:
            {"tasks": [...], "count": int}
        """
        if self.app is None or self.app.scheduler is None:
            return {"tasks": [], "count": 0}
        all_tasks = self.app.scheduler.list_tasks()
        owner_tasks = [t for t in all_tasks if t.owner == owner]
        tasks_data = []
        for task in owner_tasks:
            tasks_data.append(self._serialize_task_info(task))
        return {"tasks": tasks_data, "count": len(tasks_data)}

    async def create_scheduler_task(
        self,
        *,
        owner: str,
        schedule: dict[str, Any],
        persist: bool = True,
        misfire_policy: str = "skip",
        note: str | None = None,
    ) -> dict[str, Any]:
        """创建一个定时任务.

        Args:
            owner: 任务所有者 session_id.
            schedule: {"kind": "cron"|"interval"|"one_shot", "spec": {...}}.
            persist: 是否持久化.
            misfire_policy: misfire 处理策略.
            note: 可选的提醒内容.

        Returns:
            完整的 task info dict，包含 task_id.

        Raises:
            ValueError: schedule 格式错误.
        """
        import uuid
        from ..scheduler.contracts import CronSchedule, IntervalSchedule, OneShotSchedule

        if self.app is None or self.app.scheduler is None:
            raise RuntimeError("scheduler not available")

        # Parse schedule
        kind = str(schedule.get("kind") or "").strip()
        spec = schedule.get("spec", {})
        if not isinstance(spec, dict):
            raise ValueError("schedule.spec must be an object")

        if kind == "cron":
            expr = str(spec.get("expr") or "").strip()
            if not expr:
                raise ValueError("schedule.spec.expr is required for kind=cron")
            schedule_obj = CronSchedule(cron_expr=expr)
        elif kind == "interval":
            seconds = float(spec.get("seconds", 0))
            if seconds <= 0:
                raise ValueError("schedule.spec.seconds must be positive")
            schedule_obj = IntervalSchedule(seconds=seconds)
        elif kind == "one_shot":
            fire_at = float(spec.get("fire_at", 0))
            if fire_at <= 0:
                raise ValueError("schedule.spec.fire_at must be positive")
            schedule_obj = OneShotSchedule(fire_at=fire_at)
        else:
            raise ValueError(f"unknown schedule kind: {kind!r}")

        task_id = uuid.uuid4().hex
        metadata = {"note": note} if note else {}

        # HTTP API doesn't provide dispatcher callback - tasks created via HTTP API
        # are informational only (WEBUI-01/02), no notification on fire
        # For WEBUI-created tasks, we pass callback=None
        await self.app.scheduler.register(
            task_id=task_id,
            owner=owner,
            schedule=schedule_obj,
            callback=None,  # No notification via post_notification for HTTP-created tasks
            persist=persist,
            misfire_policy=misfire_policy,
            metadata=metadata,
        )

        # Get the created task info
        all_tasks = self.app.scheduler.list_tasks()
        task_info = next((t for t in all_tasks if t.task_id == task_id), None)
        if task_info is None:
            raise RuntimeError("task created but not found")
        return self._serialize_task_info(task_info)

    async def cancel_scheduler_task(self, *, task_id: str, owner: str) -> dict[str, Any]:
        """取消一个定时任务.

        Args:
            task_id: 待取消的任务 ID.
            owner: 任务所有者（用于验证）.

        Returns:
            {"message": "Task cancelled", "task_id": str}

        Raises:
            ValueError: task 不存在或不属于该 owner.
        """
        if self.app is None or self.app.scheduler is None:
            raise RuntimeError("scheduler not available")

        # 先验证 task 存在且属于 owner
        all_tasks = self.app.scheduler.list_tasks()
        task = next((t for t in all_tasks if t.task_id == task_id), None)
        if task is None:
            raise ValueError(f"task not found: {task_id}")
        if task.owner != owner:
            raise ValueError(f"task {task_id} does not belong to owner {owner}")

        success = await self.app.scheduler.cancel(task_id)
        if not success:
            raise RuntimeError(f"failed to cancel task: {task_id}")
        return {"message": "Task cancelled", "task_id": task_id}

    def _serialize_task_info(self, task) -> dict[str, Any]:
        """把 ScheduledTaskInfo 序列化为 JSON-safe dict."""
        from ..scheduler.contracts import CronSchedule, IntervalSchedule, OneShotSchedule

        if isinstance(task.schedule, CronSchedule):
            schedule_data = {"kind": "cron", "spec": {"expr": task.schedule.cron_expr}}
        elif isinstance(task.schedule, IntervalSchedule):
            schedule_data = {"kind": "interval", "spec": {"seconds": task.schedule.seconds}}
        elif isinstance(task.schedule, OneShotSchedule):
            schedule_data = {"kind": "one_shot", "spec": {"fire_at": task.schedule.fire_at}}
        else:
            schedule_data = {"kind": "unknown", "spec": {}}

        return {
            "task_id": task.task_id,
            "owner": task.owner,
            "schedule": schedule_data,
            "persist": task.persist,
            "misfire_policy": task.misfire_policy,
            "next_fire_at": task.next_fire_at,
            "enabled": task.enabled,
            "metadata": task.metadata,
        }
    ```
