    ```python
    """tests.test_scheduler_dispatcher 测试 ScheduledMessageDispatcher."""

    import pytest
    from unittest.mock import AsyncMock, MagicMock

    from acabot.runtime.scheduler.dispatcher import ScheduledMessageDispatcher


    @pytest.fixture
    def mock_control_plane():
        cp = MagicMock()
        cp.post_notification = AsyncMock(return_value={"delivered": True})
        return cp


    @pytest.fixture
    def dispatcher(mock_control_plane):
        return ScheduledMessageDispatcher(control_plane=mock_control_plane)


    class TestScheduledMessageDispatcher:
        def test_make_callback_returns_callable(self, dispatcher):
            callback = dispatcher.make_callback(
                conversation_id="qq:user:12345",
                note="Test notification",
            )
            assert callable(callback)
            import asyncio
            assert asyncio.iscoroutinefunction(callback)

        @pytest.mark.asyncio
        async def test_callback_sends_notification(self, dispatcher, mock_control_plane):
            callback = dispatcher.make_callback(
                conversation_id="qq:user:12345",
                note="Test notification",
            )
            await callback()

            mock_control_plane.post_notification.assert_called_once_with(
                payload={
                    "conversation_id": "qq:user:12345",
                    "text": "Test notification",
                }
            )

        @pytest.mark.asyncio
        async def test_callback_with_no_note_skips_notification(self, dispatcher, mock_control_plane):
            callback = dispatcher.make_callback(
                conversation_id="qq:user:12345",
                note=None,
            )
            await callback()

            mock_control_plane.post_notification.assert_not_called()

        @pytest.mark.asyncio
        async def test_callback_error_handling(self, dispatcher, mock_control_plane):
            mock_control_plane.post_notification.side_effect = RuntimeError("notification failed")

            callback = dispatcher.make_callback(
                conversation_id="qq:user:12345",
                note="Test",
            )
            # Should not raise, just log the error
            await callback()

            mock_control_plane.post_notification.assert_called_once()

        @pytest.mark.asyncio
        async def test_callback_captures_correct_conversation_id(self, dispatcher, mock_control_plane):
            callback = dispatcher.make_callback(
                conversation_id="qq:group:987654",
                note="Group notification",
            )
            await callback()

            mock_control_plane.post_notification.assert_called_once_with(
                payload={
                    "conversation_id": "qq:group:987654",
                    "text": "Group notification",
                }
            )
    ```
