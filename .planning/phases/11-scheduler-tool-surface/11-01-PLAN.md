    ```python
    """scheduler.dispatcher 提供定时任务回调工厂.

    Phase 3a decision D-13: callback 闭包只捕获 conversation_id (string),
    不捕获 ToolExecutionContext。fire-time 通过 control_plane.post_notification()
    重建回调上下文。
    """

    from __future__ import annotations

    import logging
    from collections.abc import Awaitable, Callable

    from acabot.runtime.control.control_plane import RuntimeControlPlane

    logger = logging.getLogger("acabot.runtime.scheduler.dispatcher")


    class ScheduledMessageDispatcher:
        """定时任务回调工厂.

        在任务创建时生成 async callback，callback 只捕获 conversation_id 和 note。
        fire-time 通过 post_notification() 发送通知到目标会话。
        """

        def __init__(
            self,
            control_plane: RuntimeControlPlane,
        ) -> None:
            """初始化 dispatcher.

            Args:
                control_plane: RuntimeControlPlane 实例，用于 fire-time 发送通知.
            """
            self._control_plane = control_plane

        def make_callback(
            self,
            conversation_id: str,
            note: str | None,
        ) -> Callable[[], Awaitable[None]]:
            """创建异步回调函数.

            Args:
                conversation_id: 目标会话的 canonical conversation_id.
                note: 可选的提醒内容，任务触发时发送给用户.

            Returns:
                一个 async callable，无参数无返回值.
            """
            async def callback() -> None:
                if not note:
                    logger.debug("Task fired with no note, skipping notification: conversation_id=%s", conversation_id)
                    return
                try:
                    await self._control_plane.post_notification(
                        payload={
                            "conversation_id": conversation_id,
                            "text": note,
                        }
                    )
                    logger.info("Scheduled task notification sent: conversation_id=%s", conversation_id)
                except Exception:
                    logger.exception("Failed to send scheduled task notification: conversation_id=%s", conversation_id)

            return callback
    ```

    ```python
    """scheduler builtin tool surface.

    单个 scheduler tool + action enum (create/list/cancel)。
    遵循 OpenClaw 风格 + Phase 3a decision D-06/D-07/D-08。
    """

    from __future__ import annotations

    import json
    import uuid
    from typing import Any

    from acabot.runtime.scheduler import (
        CronSchedule,
        IntervalSchedule,
        OneShotSchedule,
        RuntimeScheduler,
        ScheduleType,
    )
    from acabot.runtime.scheduler.dispatcher import ScheduledMessageDispatcher

    from ..tool_broker import ToolBroker, ToolExecutionContext, ToolResult

    BUILTIN_SCHEDULER_TOOL_SOURCE = "builtin:scheduler"


    class BuiltinSchedulerToolSurface:
        """scheduler builtin tool 注册入口."""

        def __init__(
            self,
            scheduler: RuntimeScheduler,
            dispatcher: ScheduledMessageDispatcher,
        ) -> None:
            """初始化 scheduler tool surface.

            Args:
                scheduler: RuntimeScheduler 实例.
                dispatcher: ScheduledMessageDispatcher 实例.
            """
            self._scheduler = scheduler
            self._dispatcher = dispatcher

        def register(self, tool_broker: ToolBroker) -> list[str]:
            """把 scheduler builtin tool 注册到 ToolBroker."""
            tool_broker.unregister_source(BUILTIN_SCHEDULER_TOOL_SOURCE)
            tool_broker.register_tool(
                self._tool_spec(),
                self._handle_scheduler,
                source=BUILTIN_SCHEDULER_TOOL_SOURCE,
            )
            return ["scheduler"]

        @staticmethod
        def _tool_spec() -> ToolSpec:
            return ToolSpec(
                name="scheduler",
                description=(
                    "创建、查看或取消定时任务。任务会在指定时间向当前会话发送通知。\n"
                    "支持三种调度类型：cron（cron 表达式）、interval（固定间隔秒数）、one_shot（一次性时间戳）。\n"
                    "创建后任务持久化到数据库，重启后继续执行。"
                ),
                parameters={
                    "type": "object",
                    "properties": {
                        "action": {
                            "type": "string",
                            "enum": ["create", "list", "cancel"],
                            "description": "操作类型：create（创建）、list（列表）、cancel（取消）",
                        },
                        "schedule": {
                            "type": "object",
                            "description": "调度配置，仅 action=create 时需要",
                            "properties": {
                                "kind": {
                                    "type": "string",
                                    "enum": ["cron", "interval", "one_shot"],
                                    "description": "调度类型",
                                },
                                "spec": {
                                    "type": "object",
                                    "description": "调度规格",
                                },
                            },
                            "required": ["kind", "spec"],
                        },
                        "note": {
                            "type": "string",
                            "description": "可选的提醒内容，任务触发时发送给用户",
                        },
                        "task_id": {
                            "type": "string",
                            "description": "要取消的任务 ID，仅 action=cancel 时需要",
                        },
                    },
                    "required": ["action"],
                },
            )

        async def _handle_scheduler(
            self,
            arguments: dict[str, Any],
            ctx: ToolExecutionContext,
        ) -> ToolResult:
            action = str(arguments.get("action", "") or "").strip()
            if action == "create":
                return await self._handle_create(arguments, ctx)
            if action == "list":
                return self._handle_list(arguments, ctx)
            if action == "cancel":
                return self._handle_cancel(arguments, ctx)
            raise ValueError(f"unsupported scheduler action: {action!r}")

        async def _handle_create(
            self,
            arguments: dict[str, Any],
            ctx: ToolExecutionContext,
        ) -> ToolResult:
            schedule_raw = arguments.get("schedule")
            if not schedule_raw:
                raise ValueError("schedule is required for action=create")
            schedule_obj = self._parse_schedule(schedule_raw)

            note = arguments.get("note")
            task_id = uuid.uuid4().hex
            # owner = conversation_id (session/channel scope)，从 ctx.target 推导
            conversation_id = self._extract_conversation_id(ctx)
            owner = conversation_id  # 绑定到触发会话的 session/channel scope

            callback = self._dispatcher.make_callback(
                conversation_id=conversation_id,
                note=note,
            )

            await self._scheduler.register(
                task_id=task_id,
                owner=owner,
                schedule=schedule_obj,
                callback=callback,
                persist=True,
                misfire_policy="skip",
                metadata={"note": note} if note else {},
            )

            response = {
                "task_id": task_id,
                "owner": owner,
                "action": "created",
                "schedule": self._serialize_schedule(schedule_obj),
            }
            return ToolResult(llm_content=json.dumps(response, ensure_ascii=False))

        def _handle_list(
            self,
            arguments: dict[str, Any],
            ctx: ToolExecutionContext,
        ) -> ToolResult:
            owner = self._extract_conversation_id(ctx)  # conversation_id 作为 owner 过滤
            all_tasks = self._scheduler.list_tasks()
            owner_tasks = [t for t in all_tasks if t.owner == owner]

            tasks_data = []
            for task in owner_tasks:
                tasks_data.append({
                    "task_id": task.task_id,
                    "owner": task.owner,
                    "schedule": self._serialize_schedule(task.schedule),
                    "persist": task.persist,
                    "misfire_policy": task.misfire_policy,
                    "next_fire_at": task.next_fire_at,
                    "enabled": task.enabled,
                    "metadata": task.metadata,
                })

            response = {"tasks": tasks_data, "count": len(tasks_data)}
            return ToolResult(llm_content=json.dumps(response, ensure_ascii=False))

        async def _handle_cancel(
            self,
            arguments: dict[str, Any],
            ctx: ToolExecutionContext,
        ) -> ToolResult:
            task_id = str(arguments.get("task_id") or "").strip()
            if not task_id:
                raise ValueError("task_id is required for action=cancel")

            success = await self._scheduler.cancel(task_id)

            if success:
                response = {"action": "cancelled", "task_id": task_id}
            else:
                response = {"action": "not_found", "task_id": task_id}
            return ToolResult(llm_content=json.dumps(response, ensure_ascii=False))

        def _parse_schedule(self, raw: Any) -> ScheduleType:
            if not isinstance(raw, dict):
                raise ValueError("schedule must be an object")
            kind = str(raw.get("kind") or "").strip()
            spec = raw.get("spec", {})
            if not isinstance(spec, dict):
                raise ValueError("schedule.spec must be an object")

            if kind == "cron":
                expr = str(spec.get("expr") or "").strip()
                if not expr:
                    raise ValueError("schedule.spec.expr is required for kind=cron")
                return CronSchedule(cron_expr=expr)
            if kind == "interval":
                seconds = spec.get("seconds")
                if seconds is None:
                    raise ValueError("schedule.spec.seconds is required for kind=interval")
                try:
                    seconds = float(seconds)
                except (TypeError, ValueError):
                    raise ValueError(f"schedule.spec.seconds must be numeric, got {seconds!r}")
                if seconds <= 0:
                    raise ValueError(f"schedule.spec.seconds must be positive, got {seconds}")
                return IntervalSchedule(seconds=seconds)
            if kind == "one_shot":
                fire_at = spec.get("fire_at")
                if fire_at is None:
                    raise ValueError("schedule.spec.fire_at is required for kind=one_shot")
                try:
                    fire_at = float(fire_at)
                except (TypeError, ValueError):
                    raise ValueError(f"schedule.spec.fire_at must be numeric, got {fire_at!r}")
                if fire_at <= 0:
                    raise ValueError(f"schedule.spec.fire_at must be positive, got {fire_at}")
                return OneShotSchedule(fire_at=fire_at)

            raise ValueError(f"unknown schedule kind: {kind!r}")

        @staticmethod
        def _serialize_schedule(schedule: ScheduleType) -> dict[str, Any]:
            if isinstance(schedule, CronSchedule):
                return {"kind": "cron", "spec": {"expr": schedule.cron_expr}}
            if isinstance(schedule, IntervalSchedule):
                return {"kind": "interval", "spec": {"seconds": schedule.seconds}}
            if isinstance(schedule, OneShotSchedule):
                return {"kind": "one_shot", "spec": {"fire_at": schedule.fire_at}}
            raise TypeError(f"Unknown schedule type: {type(schedule)}")

        @staticmethod
        def _extract_conversation_id(ctx: ToolExecutionContext) -> str:
            """从 ctx.target (EventSource) 推导 canonical conversation_id.

            注意：ctx.thread_id 是 runtime 内部执行线程标识，不参与权限模型。
            owner（session/channel scope）从 ctx.target 推导，这才是 scheduler 权限边界。
            """
            if ctx.target.message_type == "group":
                return f"{ctx.target.platform}:group:{ctx.target.group_id}"
            return f"{ctx.target.platform}:user:{ctx.target.user_id}"
    ```
